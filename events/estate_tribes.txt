#Tensions between Nomads and Sedentary Population
country_event = {
	id = tribal_estate_events.1
	title = tribal_estate_events.1.t
	desc = tribal_estate_events.1.d
	picture = NOMAD_ESTATE_eventPicture
	
	trigger = {
		has_estate = estate_nomadic_tribes
		has_estate = sedentary_subdjects
		num_of_cities = 5
		estate_territory = { 
			estate = estate_nomadic_tribes
			territory = 1
		}
		any_owned_province = {
			can_spawn_rebels = nationalist_rebels
			is_territory = no
			is_city = yes
			area_for_scope_province = {
				type = all
				is_capital = no
			}
		}
	}

	is_triggered_only = yes
	
	immediate = {
		hidden_effect = {
			random_owned_province = {
				limit = {
					can_spawn_rebels = nationalist_rebels
					is_territory = no
					is_city = yes
					area_for_scope_province = {
						type = all
						is_capital = no
					}
				}
				area = {
					limit = {
						can_spawn_rebels = nationalist_rebels
						owned_by = ROOT
						is_city = yes
					}
					set_province_flag = conflict_regarding_tax_collection
				}
			}
		}
	}
	
	mean_time_to_happen = {
		days = 1
	}
	
	option = {
		name = tribal_estate_events.1.a #	Reprimand the Chiefs.
		ai_chance = {
			factor = 75
			modifier = {
				factor = 0.5
				NOT = {
					estate_loyalty = {
						estate = estate_nomadic_tribes
						loyalty = 40
					}
				}
			}
			modifier = {
				factor = 1.5
				estate_influence = {
					estate = estate_nomadic_tribes
					influence = 65
				}
			}
		}
		add_estate_loyalty = {
			estate = sedentary_subdjects
			loyalty = 10
		}
		add_estate_loyalty = {
			estate = estate_nomadic_tribes
			loyalty = -15
		}
		add_estate_influence_modifier = {
			estate = estate_nomadic_tribes
			desc = EST_VAL_TRIBES_REPRIMANDED
			influence = -10
			duration = 5475
		}
	}
	option = {
		name = tribal_estate_events.1.b #	Let it Slide
		ai_chance = {
			factor = 25
			modifier = {
				factor = 1.5
				NOT = {
					estate_loyalty = {
						estate = estate_nomadic_tribes
						loyalty = 40
					}
				}
			}
			modifier = {
				factor = 0.5
				estate_influence = {
					estate = estate_nomadic_tribes
					influence = 65
				}
			}
		}
		random_owned_province = {
			limit = {
				has_province_flag = conflict_regarding_tax_collection
			}
			spawn_rebels = {
				size = 1
				type = nationalist_rebels
			}
			clr_province_flag = conflict_regarding_tax_collection
		}
		random_owned_province = {
			limit = {
				has_province_flag = conflict_regarding_tax_collection
				owner = { num_of_cities = 10 }
			}
			spawn_rebels = {
				size = 1
				type = nationalist_rebels
			}
			clr_province_flag = conflict_regarding_tax_collection
		}
		random_owned_province = {
			limit = {
				has_province_flag = conflict_regarding_tax_collection
				owner = { num_of_cities = 15 }
			}
			spawn_rebels = {
				size = 1
				type = nationalist_rebels
			}
			clr_province_flag = conflict_regarding_tax_collection
		}
		every_owned_province = {
			clr_province_flag = conflict_regarding_tax_collection
		}
		add_estate_loyalty = {
			estate = sedentary_subdjects
			loyalty = -10
		}
		add_estate_loyalty = {
			estate = estate_nomadic_tribes
			loyalty = 15
		}
		add_estate_influence_modifier = {
			estate = estate_nomadic_tribes
			desc = EST_VAL_TRIBES_ALLOWED_FREE_REIGN
			influence = 10
			duration = 5475
		}
	}
}

country_event = {
	id = tribal_estate_events.11
	title = tribal_estate_events.11.t
	desc = tribal_estate_events.11.d
	picture = NOMAD_ESTATE_eventPicture
	
	trigger = {
		has_estate = estate_nomadic_tribes
		has_estate = nomadic_aristocracy
		has_regency = no
		has_government_attribute = heir
		OR = {
			estate_influence = {
				estate = estate_nomadic_tribes
				influence = 70
			}
			estate_influence = {
				estate = nomadic_aristocracy
				influence = 70
			}
		}
		NOT = {
			has_estate_influence_modifier = {
				modifier = EST_VAL_KINGMAKERS
				estate = estate_nomadic_tribes
			}
		}
		NOT = {
			has_estate_influence_modifier = {
				modifier = EST_VAL_REFUSED_TO_ALTER_SUCCESSION
				estate = estate_nomadic_tribes
			}
		}
	}
	
	is_triggered_only = yes
	
	mean_time_to_happen = {
		days = 1
	}
	
	option = {
		name = tribal_estate_events.11.a
		ai_chance = {
			factor = 1
		}
		if = {
			limit = { has_heir = yes }
			kill_heir = { allow_new_heir = no }
		}
		define_heir = {
			dynasty = ROOT
			claim = 15
			age = 45
		}
		if = {
			limit = {
				estate_influence = {
					estate = nomadic_aristocracy
					influence = 70
				}
			}
			add_estate_influence_modifier = {
				desc = EST_VAL_KINGMAKERS 
				estate = nomadic_aristocracy
				influence = 10
				duration = 3650
			}
		}
		else = {
			add_estate_influence_modifier = {
				desc = EST_VAL_KINGMAKERS
				estate = estate_nomadic_tribes
				influence = 10
				duration = 3650
			}
		}
	}

		name = tribal_estate_events.11.b

					}
				}
			}
		}
		if = {
			limit = {
				estate_influence = {
					estate = nomadic_aristocracy
					influence = 70
				}
			}
			add_estate_loyalty = {
				estate = nomadic_aristocracy
				loyalty = -10
			}
			add_estate_influence_modifier = {
				desc = EST_VAL_REFUSED_TO_ALTER_SUCCESSION
