		}
		2972 = {
			owned_by = ROOT
		}
	}

	mean_time_to_happen = {
		months = 12
		modifier = {
			factor = 0.5
			is_year = 1445
		}
		modifier = {
			factor = 0.5
			is_year = 1446
		}
	}

	option = { #Send in the Army.
		name = ME_Cologne_Events.3.a
		
		ai_chance = { factor = 0.6 }
		
		custom_tooltip = ME_Colognian_Tooltips.2
		set_country_flag = Cologne_Westphalia
		add_yearly_manpower = -2
		82 = {
			add_devastation = 50
		}
		swap_non_generic_missions = yes
		country_event = { id = ME_Cologne_Events.4 }
	}
	
	option = { #We are better off giving up the territories
		name = ME_Cologne_Events.3.b
		
		ai_chance = { factor = 0.4 }
		
		set_country_flag = Cologne_Rheinland
		custom_tooltip = ME_Colognian_Tooltips.1
		swap_non_generic_missions = yes
		hidden_effect = {
			82 = {
				add_core = SOE
			}
			release = SOE
			SOE = {
				change_government = theocracy
				add_government_reform = leading_clergy_reform
			}
			if = {
				limit = {
					exists = KLE
					KLE = { is_subject = no }
				}
				KLE = {
					create_subject = {
						subject_type = vassal
						subject = SOE
					}
				}
			}
			2972 = {
				add_core = PAD
			}
			release = PAD
			PAD = {
				change_government = theocracy
				add_government_reform = leading_clergy_reform
			}
		}
		add_adm_power = 100
		add_dip_power = 100
		add_mil_power = 100
		add_stability_or_adm_power = yes
	}
}
