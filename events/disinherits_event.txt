namespace = disinherits

country_event = {
	id = disinherits.1
	title = disinherits.1.t
	desc = disinherits.1.d
	picture = COURT_eventPicture
	
	hidden = yes
	
	mean_time_to_happen = {
		months = 6
	}
	
	trigger = {
		ai = yes
		
		OR = {
			NOT = { ruler_age = 41 }
			NOT = { religion_group = christian }
		}
		OR = {
			prestige = 25
			AND = {
				prestige = 0
				NOT = { num_of_subjects = 1 }
			}
		}
		
		has_heir = yes
		has_regency = no
		is_lesser_in_union = no
		government = monarchy
		NOT = { has_reform = elective_monarchy }
		NOT = { has_reform = mamluk_government }
		NOT = { has_reform = ottoman_government }
		has_government_attribute = heir
		
		variable_arithmetic_trigger = {
			export_to_variable = {
				which = hAdm
				value = heir_adm
			}
			export_to_variable = {
				which = hDip
				value = heir_dip
			}
			export_to_variable = {
				which = hMil
				value = heir_mil
			}
			change_variable = {
				which = hAdm
				which = hDip
			}
			change_variable = {
				which = hAdm
				which = hMil
			}
			NOT = {
				check_variable = { hAdm = 7 }
			}
		}
	}
	
	option = {
		name = disinherits.1.a
		
		log = "[This.GetName] disinherits"
		add_prestige = -50
		remove_heir = yes
	}
}
