	hidden = yes
	is_triggered_only = yes
	
	trigger = {
		is_subject_of_type = imperial_feud_subject

		overlord = {
			NOT = {
				is_in_war = {
					participants = FROM
					participants = THIS
				}
			}
		}
		overlord = {
			reverse_has_opinion_modifier = {
				modifier = refused_join_war
				who = ROOT
			}
		}
	}
	
	immediate = {
		hidden_effect = {
			overlord = {
				every_country = {
					limit = {
						NOT = { war_with = PREV }
						NOT = { war_with = ROOT }
						OR = {
							is_part_of_hre = yes
							AND = {
								is_subject_of = PREV
								is_subject_of_type = imperial_feud_subject
							}
						}
					}
					add_opinion = {
						who = PREV
						modifier = refused_to_defend_empire
					}
				}
			}
		}
	}
	
	option = {
		name = subjects_expanded_imperial_feud_subject_events6.EVTOPT1
		
		ai_chance = { factor = 100 }
	}
}
