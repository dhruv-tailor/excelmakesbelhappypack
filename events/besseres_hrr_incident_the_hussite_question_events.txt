country_event = {

	desc = besseres_hrr_incident_the_hussite_question_events.EVTDESC1
	picture = HRE_eventPicture

	fire_only_once = yes

	trigger = {
		has_dlc = "Art of War"
		is_emperor = yes
		hre_religion_treaty = no
		hre_religion_locked = yes
		OR = {
			hre_religion = protestant
			hre_religion = reformed
		}
		NOT = { hre_reform_passed = erbkaisertum }
		NOT = { hre_reform_passed = emperor_erbkaisertum }
		NOT = { hre_reform_passed = landfriede }
		NOT = { hre_reform_passed = emperor_landfriede }
		OR = {
			NOT = { has_dlc = "Emperor" }
			NOT = { active_imperial_incident = any }
		}
		any_country = {
			religion = hussite
			is_part_of_hre = yes
			is_subject = no
			government = monarchy
			has_regency = no
		}
		OR = {
			any_country = {
				religion = hussite
				is_part_of_hre = yes
				is_free_or_tributary_trigger = yes
				total_development = 500
			}
			AND = {
				any_elector = {
					religion = hussite
				}
				calc_true_if = {
					all_country = {
						religion = hussite
						is_part_of_hre = yes
					}
					amount = 3
				}
			}
			calc_true_if = {
				all_country = {
					religion = hussite
					is_part_of_hre = yes
				}
				amount = 5
			}
		}
	}

	mean_time_to_happen = {
		months = 60
	}

	option = {
		name = besseres_hrr_incident_the_hussite_question_events1.EVTOPT1

		if = {
			limit = {
				has_dlc = "Emperor"
			}
			set_global_flag = hre_the_hussite_question_incident_active
			set_imperial_incident = besseres_hrr_incident_the_hussite_question
		}
		else = {
			country_event = { id = besseres_hrr_incident_the_hussite_question_events.2 }
		}
	}
}

country_event = {
	id = besseres_hrr_incident_the_hussite_question_events.5

	desc = besseres_hrr_incident_the_hussite_question_events.EVTDESC5
	picture = HRE_eventPicture

	major = yes
	is_triggered_only = yes

	immediate = {
		hidden_effect = {
			set_hre_religion_locked = no
			set_hre_religion_treaty = yes
			every_elector = {
				elector = no
				if = {
					limit = {
						religion = hussite
					}
					set_country_flag = was_hussite_elector
				}
			}
			FROM = {
				elector = yes
				if = {
					limit = {
						has_country_flag = was_hussite_elector
					}
					clr_country_flag = was_hussite_elector
				}
			}
			if = {
				limit = {
					religion = reformed
				}
				set_hre_heretic_religion = reformed
			}
			else = {
				set_hre_heretic_religion = protestant
			}
			set_hre_religion = hussite
			set_hre_religion_locked = yes
			emperor = {
				if = {
					limit = {
						NOT = { religion = hussite }
					}
					if = {
						limit = {
							has_heir = yes
						}
						set_country_flag = exiled_non_hussite_emperor_heir
						exile_heir_as = exiled_non_hussite_emperor_heir
					}
					define_heir = {
						dynasty = THIS
						male = yes
						age = 20
						claim = 100
					}
					exile_ruler_as = exiled_non_hussite_emperor_ruler

						limit = {
							has_country_flag = exiled_non_hussite_emperor_heir
						}
						clr_country_flag = exiled_non_hussite_emperor_heir
						set_heir = exiled_non_hussite_emperor_heir
					}
				}
			}
			every_country = {
				limit {
					has_country_flag = was_hussite_elector
				}
				clr_country_flag = was_hussite_elector
				if = {
					limit = {
						NOT = { num_of_electors = 7 }
						is_elector = no
					}
					elector = yes
				}
			}
			every_country = {
				limit = {
					religion = hussite
					is_part_of_hre = yes
					is_elector = no
				}
				if = {
					limit = {
						NOT = { num_of_electors = 7 }
					}
					elector = yes
				}
			}
		}
	}

	option = {
