# Refused ceremonial burial of relative
country_event = {
	id = fetishist_flavor.5
	title = fetishist_flavor.5.t
	picture = PRAYING_eventPicture

	trigger = {
		OR = {
			has_adopted_cult = yemoja_cult
			has_adopted_cult = roog_cult
			has_adopted_cult = nyame_cult
		}

		has_regency = no
		ruler_age = 15

		any_neighbor_country = {
			NOT = { war_with = ROOT }
			NOT = { alliance_with = ROOT }
			NOT = {
				has_opinion = {
					who = ROOT
					value = 50
				}
			}
		}
	}
	immediate = {
		hidden_effect = {
			random_neighbor_country = {
				limit = {
					NOT = { war_with = ROOT }
					NOT = { alliance_with = ROOT }
					NOT = {
						has_opinion = {
							who = ROOT
							value = 50
						}
					}
				}
				save_event_target_as = nonchalant_neighbor
			}
			set_ruler_flag = upset_by_tragically_killed_cousin
		}
	}
	desc = {
		trigger = { has_adopted_cult = yemoja_cult }
		desc = fetishist_flavor.5.d.a
	}
	desc = {
		trigger = { has_adopted_cult = roog_cult }
		desc = fetishist_flavor.5.d.b
	}
	desc = {
		trigger = { has_adopted_cult = nyame_cult }
		desc = fetishist_flavor.5.d.c
	}
	desc = {
		trigger = { 
			NOT = { has_adopted_cult = yemoja_cult }
			NOT = { has_adopted_cult = roog_cult }
			NOT = { has_adopted_cult = nyame_cult }
		}
		desc = fetishist_flavor.5.d.d
	}

	mean_time_to_happen = {
		months = 400
	}

	option = { #West African: # Pay for burial yourself
		name = fetishist_flavor.5.a
		custom_tooltip = fetishist_flavor.5.a.tt

		ai_chance = { factor = 20 }

		add_treasury = -35
		add_dip_power = 60
		country_event = { id = fetishist_flavor.7 days = 20 }

		event_target:nonchalant_neighbor = {
			add_opinion = {
				who = ROOT
				modifier = opinion_approves_of_diplomacy
			}
		}

		trigger = { 
			OR = {
				has_adopted_cult = yemoja_cult
				has_adopted_cult = roog_cult
				has_adopted_cult = nyame_cult
			}
		}
	}
	option = { # Who do they think they are??
		name = fetishist_flavor.5.b
		custom_tooltip = fetishist_flavor.5.b.tt

		ai_chance = { factor = 80 }

		add_opinion = {
			who = event_target:nonchalant_neighbor
			modifier = opinion_disapproves_of_threats
		}
		add_casus_belli = {
			target = event_target:nonchalant_neighbor
			type = cb_insult
			months = 12
		}
		event_target:nonchalant_neighbor = {
			country_event = { id = fetishist_flavor.6 days = 1 }
		}
	}
	option = { # Let it slide
		name = fetishist_flavor.5.c
		custom_tooltip = fetishist_flavor.5.c.tt

		event_target:nonchalant_neighbor = {
			add_opinion = {
				who = ROOT
				modifier = opinion_approves_of_diplomacy
			}
		}
	}
}
