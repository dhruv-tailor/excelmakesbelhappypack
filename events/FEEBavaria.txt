#Bavarian Events

namespace = fee_BAV


# Dealing With the Robber Barons

country_event = {
	id = fee_BAV.1
	title = fee_BAV.1.t
	desc = fee_BAV.1.d
	picture = BORDER_TENSION_eventPicture

	fire_only_once = yes

	trigger = {
		tag = BAV
		NOT = { is_year = 1555}
		owns_core_province = 65
	}

	mean_time_to_happen = {
		months = 120
	}

	option = {
		name = fee_BAV.1.a
		add_years_of_income = -0.25
		add_manpower = -2
		add_country_modifier = {
			name = fee_fought_robber_barons
			duration = 3650
		}
	}

	option = {
		name = fee_BAV.1.b
		add_treasury = 25
	}
}


#One Duchy, United

country_event = {
	id = fee_BAV.2
	title = fee_BAV.2.t
	desc = fee_BAV.2.d
	picture = CONQUEST_eventPicture
	
	fire_only_once = yes
	
	trigger = {
		tag = BAV
		is_year = 1490
		all_owned_province = {
			OR = {
				AND = {
					culture = bavarian
					NOT = {
						local_autonomy = 10
					}
				}
				NOT = { culture = bavarian }
			}
		}
	}
	
	mean_time_to_happen = {
		months = 60
	}
	
	option = {
		name = fee_BAV.2.a
		add_stability_or_adm_power = yes
		add_legitimacy = 25
	}
	
}

# Reconciliation with the Habsburgs

country_event = {
	id = fee_BAV.3
	title = fee_BAV.3.t
	desc = fee_BAV.3.d
	picture = DIPLOMACY_eventPicture
	
	fire_only_once = yes
	
	trigger = {
		tag = BAV
		is_year = 1520
		NOT = {
			is_year = 1600
		}
		is_at_war = no
		NOT = {
			is_rival = HAB
		}
		is_lesser_in_union = no
		dynasty = "von Wittelsbach"
		HAB = {
			exists = yes
			is_emperor = yes
			dynasty =  "von Habsburg"
		}
	}
	
	option = {	
		name = fee_BAV.3.a	
		HAB = {
			add_opinion = {
				who = ROOT
				modifier = fee_bav_reconcile_habsburgs
			}
		}
	}

	option = {	
		name = fee_BAV.3.b	
		HAB = {
			add_opinion = {
				who = ROOT
				modifier = fee_bav_against_habsburgs
			}
		}
		add_legitimacy = 5
		add_prestige = 5
	}
}
