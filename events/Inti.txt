#Birth of a Panaca
country_event = {
	id = inti_events.1
	title = inti_events.1.t
	desc = inti_events.1.d
	picture = INCAN_GODS_eventPicture
	
	trigger = {
		religion = inti
		has_dlc = "El Dorado"
		is_religion_reformed = no
		has_regency = no
		has_new_dynasty = no
	}
	
	is_triggered_only = yes
	
	option = {
		name = inti_events.1.a #XXXs Panaca is born.
		custom_tooltip = inti_events.1.a.tt #The Mummy takes it's place as the head of a new Panaca, taking it's property with it.
		add_authority = 10
		add_years_of_income = -0.5
	}
	option = {
		name = inti_events.1.b #This is a powerful Huaca.
		custom_tooltip = inti_events.1.b.tt #Rather than the head of a Panaca the Mummy will become a minor good to be worshipped.
		add_authority = -10
		add_prestige = 10
	}
	option = {
		name = inti_events.1.c #This ruler will be forgotten
		custom_tooltip = inti_events.1.c.tt #None but his own relatives shall be allowed to remember our predecessor, his property will be seized and in a few generations most will not remember of his existence.
		add_years_of_income = 0.5
		add_stability = -1
		random = {
			chance = 40
			random_owned_province = {
				spawn_rebels = {
					type = pretender_rebels
					religion = ROOT
					size = 1
				}
			}
		}
	}
}

#Mitmaq Wanting to Go Home
country_event = {
	id = inti_events.17
	title = inti_events.17.t
	desc = inti_events.17.d
	picture = NATIVES_HIGH_eventPicture
	
	trigger = {
		religion = inti
		has_dlc = "El Dorado"
		is_religion_reformed = no
		has_regency = no
		num_of_cities = 5
		any_owned_province = {
			NOT = { months_since_defection = 60 }
			NOT = { has_province_flag = inti_mitmaq_gone_home }
			has_siege = no
			is_city = yes
			base_tax = 2
		}
	}
	
	immediate = {
		hidden_effect = {
			random_owned_province = {
				limit = {
					base_tax = 2
					NOT = { months_since_defection = 60 }
					NOT = { has_province_flag = inti_mitmaq_gone_home }
				}
				set_province_flag = inti_mitmaq_wants_to_go
			}
		}
	}
	
	mean_time_to_happen = {
		months = 260
	}
	
	option = {
		name = inti_events.17.a #Allow the Mitmaq to Return.
		random_owned_province = {
			limit = {
				not = { is_capital = yes }
				not = { has_province_flag = inti_mitmaq_wants_to_go }
			}
			add_base_tax = 1
		}
		random_owned_province = {
			limit = { has_province_flag = inti_mitmaq_wants_to_go }
			add_base_tax = -1
			set_province_flag = inti_mitmaq_gone_home
			clr_province_flag = inti_mitmaq_wants_to_go
		}
	}
	option = {
		name = inti_events.17.b #They are needed where they are
		random_owned_province = {
			limit = { has_province_flag = inti_mitmaq_wants_to_go }
			add_unrest = 5
			clr_province_flag = inti_mitmaq_wants_to_go
		}
	}
}
