#Seek Lithuanian Protection
country_event = {
	id = fee_rst.2
	title = fee_rst.2.t
	desc = fee_rst.2.d
	picture = DIPLOMACY_eventPicture
	
	trigger = {
		OR = {
			tag = TVE
			tag = RYA
			tag = PSK
		}
		is_subject = no
		NOT = {
			has_country_flag = lithuanian_protection
		}
		LIT = {
			is_subject = no
			NOT = {
				alliance_with = ROOT
			}
		}
		NOT = {
			num_of_cities = 5
		}
		any_neighbor_country = {
			tag = MOS
		}
		is_year = 1450
	}
   mean_time_to_happen = {
	months = 300
   }
   
   immediate = {
	hidden_effect = {
		set_country_flag = lithuanian_protection
	}
   }
   
   option = {
	name = fee_rst.2.a
		add_mil_power = 25
	}
		  
	option = {
		name = fee_rst.2.b
		add_years_of_income = -1
		if = {
			limit = {
				LIT = {
					ai = yes
				}
			}
			create_alliance = LIT
		}
		if = {
			limit = {
				LIT = {
					ai = no
				}
			}
			country_event = { id = fee_rst.3  days = 5 }
		}
	}
}
