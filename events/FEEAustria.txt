#Austrian events

namespace = fee_HAB

#The Diet of Worms
country_event = {
    id = fee_HAB.1
    title = fee_HAB.1.t
    desc = fee_HAB.1.d
    picture = HRE_eventPicture

    fire_only_once = yes

    trigger = {
        tag = HAB
        is_emperor = yes
        is_year = 1480
        NOT = {
            is_religion_enabled = protestant
        }
        NOT = { hre_reform_passed = hofgericht }
        NOT = { hre_reform_passed = emperor_reichsstabilitaet }
        imperial_influence = 20
    }

    mean_time_to_happen = {
        months = 120
    }

    option = {
        name = fee_HAB.1.a
        add_imperial_influence = 20
        add_dip_power = 20
        add_mil_power = 20
        add_stability = -1
        if = {
            limit = {
                61 = {
                    is_part_of_hre = yes
                }
            }
            61 = {
                add_base_tax = 1
            }
        }
        add_country_modifier = {
            name = fee_diet_worms
            duration = 3650
        }
    }
    option = {
        name = fee_HAB.1.b
        add_imperial_influence = -20
        every_country = {
            limit = {
                is_part_of_hre = yes
            }
            add_opinion = {
                who = HAB
                modifier = fee_diet_worms_opinion
            }
        }
    }
}

#Rise of the Ottomans in Europe
country_event = {
    id = fee_HAB.2
    title = fee_HAB.2.t
    desc = fee_HAB.2.d
    picture = HRE_eventPicture

    fire_only_once = yes

    trigger = {
        tag = HAB
        TUR = {
            num_of_cities = 12
        }
        NOT = {
            any_owned_province = {
                NOT = {
                    continent = europe
                }
            }
        }
        OR = {
            any_owned_province = {
                any_neighbor_province = {
                    owned_by = TUR
                }
            }
            any_subject_country = {
                any_owned_province = {
                    any_neighbor_province = {
                        owned_by = TUR
                    }
                }
            }
        }
    }

    mean_time_to_happen = {
        months = 6
    }

    option = {
        name = fee_HAB.2.a
        ai_chance = {
            factor = 100
        }
        add_mil_power = 50
        add_manpower = 3
        add_army_tradition = 5.0
        reverse_add_opinion = {
            who = TUR
            modifier = fee_defense_of_christendom
        }
    }
    option = {
        name = fee_HAB.2.b
        ai_chance = {
            factor = 0
        }
        add_dip_power = 25
        add_treasury = 10
    }
}

#The Austrian Baroque

	id = fee_HAB.3

	desc = fee_HAB.3.d
	picture = COURT_eventPicture

	fire_only_once = yes

	trigger = {
		tag = HAB
		is_year = 1650
		NOT = {
			is_year = 1730
		}
		government = monarchy
		num_of_cities = 10
		legitimacy = 80
		stability = 1
	}

	mean_time_to_happen = {
		months = 600
		modifier = {
			factor = 0.8
			stability = 2
		}
		modifier = {
			factor = 0.8
			legitimacy = 100
		}
	}

	option = {
		name = fee_HAB.3.a
		ai_chance = {
			factor = 85
		}
		add_country_modifier = {
			name = fee_austrian_baroque
			duration = -1
		}
		custom_tooltip = fee_HAB.3.tt
	}
	option = {
		name = fee_HAB.3.b
		ai_chance = {
			factor = 15
		}
		add_stability_or_adm_power = yes
		add_adm_power = 10
		add_dip_power = 10
	}
}

	}

				}

				factor = 0

			}

		hidden_effect = {
			add_country_modifier = {

			}
		}
		add_treasury = -50
		add_prestige = 5
		random_owned_province = {
			limit = {
				development = 16
			}
			add_base_tax = 1
		}
