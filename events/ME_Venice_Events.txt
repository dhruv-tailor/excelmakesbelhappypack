country_event = { # Establishing the Scuola degli Albanesi
	id = ME_Venice_Events.4
	title = ME_Venice_Events.4.T
	desc = ME_Venice_Events.4.D
	picture = DIPLOMACY_eventPicture

	fire_only_once = yes

	trigger = {
		tag = VEN
		OR = {
			AND = {
				exists = ALB
				ALB = {
					OR = {
						alliance_with = ROOT
						is_subject_of = ROOT
						has_opinion = {
							who = ROOT
							value = 125
						}
					}
				}
			}
			any_owned_province = {
				culture = albanian
			}
		}
	}

	mean_time_to_happen = {
		months = 12
	}

	option = {
		name = ME_Venice_Events.4.OPT1
		ai_chance = {
			factor = 50
		}

		add_prestige = -5
	}

	option = {
		name = ME_Venice_Events.4.OPT2
		ai_chance = {
			factor = 50
		}

		add_treasury = -60
		if = {
			limit = {
				has_dlc = "Mandate of Heaven"
			}
			add_splendor = 200
		}
		else = {
			add_legitimacy_equivalent = { amount = 5 republican_tradition = 3 }
		}
	}
}
