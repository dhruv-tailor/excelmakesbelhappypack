country_event = {
	id = ME_Knights_Events.3
	title = ME_Knights_Events.EVTNAME3
	desc = ME_Knights_Events.EVTDESC3
	picture = FORT_eventPicture
	
	is_triggered_only = yes

	option = {
		name = ME_Knights_Events3.EVTOPT1
		
		ai_chance = { factor = 0.25 }
		
		if = {
			limit = {
				NOT = { total_development = 100 }
			}
			add_treasury = -10
			KNI = {
				country_event = {
					id = ME_Knights_Events.4
					days = 30
					random = 330
				}
			}
		}
		else_if = {
			limit = {
				total_development = 100
				NOT = { total_development = 300 }
			}
			add_treasury = -50
			KNI = {
				country_event = {
					id = ME_Knights_Events.5
					days = 30
					random = 330
				}
			}
		}
		else = {
			add_treasury = -100
			KNI = {
				country_event = {
					id = ME_Knights_Events.6
					days = 30
					random = 330
				}
			}
		}
		if = {
			limit = {
				religion = catholic
			}
			add_papal_influence = 25
		}
	}
	
	option = {
		name = ME_Knights_Events3.EVTOPT2
		
		ai_chance = { factor = 0.25 }
	}
}
