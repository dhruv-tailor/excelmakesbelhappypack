# Titi'kaka
country_event = {
	id = flavor_aimara.2
	title = flavor_aimara.2.t
	desc = flavor_aimara.2.d
	picture = RELIGION_eventPicture

	trigger = {
		OR = {
			tag = PCJ
			tag = CRA
			tag = CLA
		}
		OR = {
			AND = {
				owns = 2947
				2947 = {
					num_free_building_slots = 1
					NOT = {
						has_building = temple
					}
				}
			}
			AND = {
				owns = 2831
				2831 = {
					NOT = {
						has_building = temple
					}
				}
			}
			AND = {
				owns = 804
				804 = {
					num_free_building_slots = 1
					NOT = {
						has_building = temple
					}
				}
			}
			AND = {
				owns = 802
				802 = {
					num_free_building_slots = 1
					NOT = {
						has_building = temple
					}
				}
			}
		}
	}
	
	mean_time_to_happen = {
		months = 200
	}
	
	option = {
		name = flavor_aimara.2.a
		add_treasury = -30
		random_owned_province = {
			limit = {
				NOT = {
					has_building = temple
				}
				num_free_building_slots = 1
				OR = {
					province_id = 2947
					province_id = 2831
					province_id = 804
					province_id = 802
				}
			}
			add_building = temple
		}
	}
	option = {
		name = flavor_aimara.2.b
		add_stability_or_adm_power = yes
	}
}

# Túpac Amaru II and Micaela Bastidas Puyucahua

	id = flavor_aimara.3
	title = flavor_aimara.3.t

	picture = BATTLE_eventPicture

		OR = {
			tag = PCJ
			tag = CRA
			tag = CLA

		is_year = 1600
		any_neighbor_country = {
			capital_scope = { continent = europe }

# The Troublesome Yatiri
country_event = {
	id = flavor_aimara.4
	title = flavor_aimara.4.t
	desc = flavor_aimara.4.d
	picture = DIPLOMACY_eventPicture

	fire_only_once = yes
	
	trigger = {
		OR = {
			tag = PCJ
			tag = CRA
			tag = CLA
		}
		OR = {
			revolt_percentage = 0.05
			any_owned_province = {
				unrest = 2
			}
		}
		is_at_war = no
	}
	
	mean_time_to_happen = {
		months = 32
	}
	
	option = {
		name = flavor_aimara.4.a
		add_prestige = -5

			limit = {
				unrest = 2
			}
			add_province_modifier = {
				name = remove_unrest_5_modifier
				duration = 1825
			}
		}
	}
	option = {
		name = flavor_aimara.4.b
		add_prestige = 5
		random_owned_province = {
			limit = {
				unrest = 2
			}
			add_province_modifier = {
				name = add_unrest_5_modifier

			}
		}
	}
}
