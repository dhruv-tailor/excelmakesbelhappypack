# Ruled by Despot
country_event = {
	id = republics.3
	title = republics.3.t
	desc = republics.3.d
	picture = DEBATE_REPUBLICAN_eventPicture
	
	trigger = {
		government = republic
		is_colonial_nation = no
		 OR = {
            NOT = { has_dlc = "Res Publica" }
            NOT = { has_reform = dutch_republic }
        }
		OR = {
			AND = {
				NOT = { republican_tradition = 40 }
				has_dlc = "Res Publica"
			}
			NOT = { republican_tradition = 1 }
		}
		NOT = {
			OR = {
				has_reform = revolutionary_republic_reform
				has_reform = pirate_republic_reform
				has_reform = pirate_king_reform
				has_reform = war_against_the_world_doctrine_reform
				has_reform = black_market_consortium_reform
			}
		}
	}
	
	mean_time_to_happen = {
		months = 240
		modifier = {
			factor = 0.5
			NOT = { republican_tradition = 30 }
		}
		modifier = {
			factor = 0.5
			NOT = { republican_tradition = 20 }
		}
		modifier = {
			factor = 0.5
			NOT = { republican_tradition = 10 }
		}		
		modifier = {
			factor = 0.01
			NOT = { republican_tradition = 1 }
		}			
	}
	
	immediate = {
		hidden_effect = {
			if = {
				limit = { is_tribal = yes }
				change_government = monarchy
				set_legacy_government = despotic_monarchy
			}
			if = {
				limit = {
					is_tribal = no  
					has_dlc = "Res Publica"
					NOT = { has_reform = revolutionary_republic_reform }
				}
				add_government_reform = presidential_despot_reform
			}
			if = {
				limit = {
					is_tribal = no  
					NOT = { has_dlc = "Res Publica" }
					NOT = { has_reform = revolutionary_republic_reform }
				}
				change_government = monarchy
				set_legacy_government = despotic_monarchy
			}
			if = {
				limit = {
					has_reform = revolutionary_republic_reform
				}
				change_government = monarchy
				add_government_reform = revolutionary_empire_reform
			}
		}
	}
	
	option = {
		name = republics.3.a
		tooltip = {
			if = {
				limit = { is_tribal = yes }
				change_government = monarchy
				set_legacy_government = despotic_monarchy
			}
			if = {
				limit = {
					is_tribal = no  
					has_dlc = "Res Publica"
				}
				add_government_reform = presidential_despot_reform
			}
			if = {
				limit = {
					is_tribal = no  
					NOT = { has_dlc = "Res Publica" }
				}
				change_government = monarchy
				set_legacy_government = despotic_monarchy
			}
		}
	}
}
