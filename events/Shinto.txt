#Wokou - Bahan
country_event = {
	id = shinto_events.1
	title = "shinto_events.1.t"
	desc = "shinto_events.1.d"
	picture = WOKOU_INCIDENT_eventPicture
	
	is_triggered_only = yes
	
	immediate = {
		hidden_effect = {
			random_neighbor_country = {
				save_event_target_as = saved_neighbor
			}
			random_known_country = { #Worst fit
				limit = {
					num_of_ports = 1
					NOT = { religion = shinto }
				}
				save_event_target_as = save_wokou_target
			}
			random_known_country = { #Better fit
				limit = {
					num_of_ports = 2
					NOT = { religion = shinto }
				}
				save_event_target_as = save_wokou_target
			}
			random_known_country = { #Even better fit
				limit = {
					num_of_ports = 5
					NOT = { religion = shinto }
				}
				save_event_target_as = save_wokou_target
			}
			random_known_country = { #Nice Fit
				limit = {
					num_of_ports = 10
					NOT = { religion = shinto }
				}
				save_event_target_as = save_wokou_target
			}
			random_known_country = { #Great Fit
				limit = {
					num_of_ports = 15
					NOT = { religion = shinto }
				}
				save_event_target_as = save_wokou_target
			}
		}
	}
	
	option = {
		name = "shinto_events.1.a" #We must control this so we can be the ones making the money from it.
		custom_tooltip = shinto_incident_wokou_start.tt
		custom_tooltip = shinto_events.open.tt
		add_adm_power = -25
		add_incident_variable_value = {
			incident = incident_wokou
			value = 2
		}
		hidden_effect = {
			country_event = {
				id = shinto_events.2
				days = 1095
			}
		}
	}
	
	option = {
		name = "shinto_events.1.b" #We must stop these Picaroons!
		custom_tooltip = shinto_events.isolation.tt
		custom_tooltip = shinto_incident_wokou_start.tt
		add_mil_power = -25
		add_dip_power = -25
		add_incident_variable_value = {
			incident = incident_wokou
			value = 3
		}
		hidden_effect = {
			country_event = {
				id = shinto_events.2
				days = 1095
			}
		}
	}
}

country_event = {
	id = shinto_events.3
	title = "shinto_events.3.t"
	desc = "shinto_events.3.d"
	picture = WOKOU_INCIDENT_eventPicture
	
	is_triggered_only = yes

	immediate = {
		hidden_effect = {
			random_owned_province = { #fallback
				limit = {
					has_port = yes
					is_in_capital_area = yes
				}
				save_event_target_as = save_wokou_target
			}
			random_owned_province = {
				limit = {
					has_port = yes
					is_in_capital_area = yes
					is_capital = no
				}
				save_event_target_as = save_wokou_target
			}
		}
	}
	
	option = {
		name = "shinto_events.3.a" #What do we care if he lives of the wealth of foreigners?
		custom_tooltip = shinto_events.open.tt
		
		event_target:save_wokou_target = {
			add_province_modifier = {
				name = wokou_pirate_stronghold_indifferent
				duration = 3650
			}
		}
		add_incident_variable_value = {
			incident = incident_wokou
			value = 1
		}
		set_country_flag = end_of_wokou_chain
	}
	
	option = {
		name = "shinto_events.3.b" #Perhaps he could be made a servant of our state...
		custom_tooltip = shinto_events.isolation.tt
		add_years_of_income = -0.5

			tradition = 50
		}
		event_target:save_wokou_target = {

				name = wokou_pirate_stronghold_friendly
				duration = 3650
			}
		}
		add_incident_variable_value = {
			incident = incident_wokou
			value = 2
		}
		set_country_flag = end_of_wokou_chain

			add_province_modifier = {
				name = wokou_pirate_stronghold_hostile
				duration = 3650
			}
		}
		add_incident_variable_value = {
			incident = incident_wokou
			value = 3
		}
		set_country_flag = end_of_wokou_chain
	}
}
