# Promising Inventor
country_event = {
	id = innovativeness_events.4
	title = "innovativeness_events.4.t"
	desc = "innovativeness_events.4.d"
	picture = SCIENTISTS_eventPicture
	
	trigger = {
		has_dlc = "Rule Britannia"
		any_neighbor_country = {
			OR = {
				adm_tech = ROOT
				dip_tech = ROOT
				mil_tech = ROOT
			}
			num_of_cities = ROOT
		}
	}
	
	is_triggered_only = yes
	
	immediate = {
		hidden_effect = {
			random_neighbor_country = {
				limit = {
					OR = {
						adm_tech = ROOT
						dip_tech = ROOT
						mil_tech = ROOT
					}
					num_of_cities = ROOT
				}
				save_event_target_as = inventor_country
			}
			set_saved_name = {
				key = inventor
				type = advisor
				scope = event_target:inventor_country
			}
		}
	}
	
	mean_time_to_happen = {
		days = 1
	}
	
	option = {
		name = "innovativeness_events.4.a" #
		ai_chance = {
			factor = 40
		}
		add_innovativeness_small_effect = yes
		event_target:inventor_country = {
			add_opinion = {
				who = ROOT
				modifier = harbored_inventor
			}
			every_ally = {
				limit = { NOT = { tag = ROOT } }
				add_opinion = {
					who = ROOT
					modifier = harbored_inventor
				}
			}
		}
	}
	option = {
		name = "innovativeness_events.4.b"
		ai_chance = {
			factor = 60
		}
		add_prestige = 10
		event_target:inventor_country = {
			add_opinion = {
				who = ROOT
				modifier = turned_away_inventor
			}
		}
	}
	after = {
		clear_saved_name = inventor
	}
}
