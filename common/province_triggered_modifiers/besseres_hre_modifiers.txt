# Modifier to trigger travel to Rome
preparations_for_travel_to_rome = {
	potential = {
		always = yes
	}

	trigger = {
		owner = { is_emperor = yes }
	}
	
	hidden = yes

	picture = legitimacy

	#Province scope
	on_activation = {
		owner = {
			country_event = { id = besseres_hrr_events.22 }
		}
	}

	on_deactivation = {
		if = {
			limit = {
				owner = { is_emperor = yes }
			}
			remove_province_triggered_modifier = preparations_for_travel_to_rome
		}
	}
}
