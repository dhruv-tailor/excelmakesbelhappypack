iv_building_1_1 = {

	potential = {
		has_global_flag = no_building_state_maintenance
		calc_true_if = {
			amount = 1
			has_building = infrastructure_lvl_1
			has_building = workshop
			has_building = temple
			has_building = marketplace
			}

		NOT = {
			calc_true_if = {
			amount = 2
			has_building = infrastructure_lvl_1
			has_building = workshop
			has_building = temple
			has_building = marketplace
			}
		}
	}

	trigger = {
		has_global_flag = no_building_state_maintenance
		calc_true_if = {
			amount = 1
			has_building = infrastructure_lvl_1
			has_building = workshop
			has_building = temple
			has_building = marketplace
			}

		NOT = {
			calc_true_if = {
			amount = 2
			has_building = infrastructure_lvl_1
			has_building = workshop
			has_building = temple
			has_building = marketplace
			}
		}
	}

	local_state_maintenance_modifier = 0.1

}
