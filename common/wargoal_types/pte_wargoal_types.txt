pte_defend_rival_subject_war_goal = {
	type = defend_country

	allowed_countries = {
		NOT = { war_with = ROOT }
		ROOT = {
			any_rival_country = {
				is_subject = no
				OR = {
					is_in_war = {
						attacker_leader = FROM
						defender_leader = THIS
						casus_belli = cb_independence_war
					}
					is_in_war = {
						attacker_leader = FROM
						defender_leader = THIS
						casus_belli = cb_colonial_independance_war
					}
				}
			}
			NOT = {
				calc_true_if = {
					all_rival_country = {
						is_subject = no
						war_with = FROM
						OR = {
							is_in_war = {
								attacker_leader = THIS
								defenders = FROM
							}
							is_in_war = {
								defender_leader = THIS
								attackers = FROM
							}
						}
					}
					amount = 2
				}
			}
		}
	}

	attacker = {
		badboy_factor = 0
		prestige_factor = 1.5
		peace_cost_factor = 0.5

		peace_options = {
			pte_destabilize_nation
			pte_army_disarmament
			po_humiliate_rival
			po_humiliate
			pte_demand_manpower
			pte_demand_manpower_mam
			pte_demilitarize_border
			po_war_reparations
			po_gold
		}
	}

	defender = {
		badboy_factor = 1
		prestige_factor = 1
		peace_cost_factor = 1

		peace_options = {
			pte_destabilize_nation
			pte_army_disarmament
			po_humiliate_rival
			po_humiliate
			pte_demand_manpower
			pte_demand_manpower_mam
			pte_demilitarize_border
			po_war_reparations
			po_gold
		}
	}

	war_name = PTE_DEFEND_RIVAL_SUBJECT_WAR_NAME
}
