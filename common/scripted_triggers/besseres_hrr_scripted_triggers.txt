num_of_max_electors = {
	variable_arithmetic_trigger = {
		custom_tooltip = NUM_OF_MAX_ELECTORS_TT_$value$
		export_to_variable = {
			which = num_of_max_electors
			value = modifier:max_electors
			who = emperor
		}
		change_variable = { which = num_of_max_electors value = 7 }
		check_variable = { which = num_of_max_electors value = $value$ }
	}
}

is_valid_elector = {
	is_subject = no
	is_part_of_hre = yes
	capital_scope = {
		continent = europe
	}
	is_emperor = no
	is_elector = no
	religion_group = christian
	NOT = { has_reform = free_city }
	NOT = { is_enemy = ROOT }
	NOT = { is_rival = ROOT }
	OR = {
		has_opinion = {
			who = ROOT
			value = 25
		}
		alliance_with = ROOT
		marriage_with = ROOT
		guaranteed_by = ROOT
	}
	OR = {
		NOT = { has_dlc = "Art of War" }
		AND = {
			has_dlc = "Art of War"
			OR = {
				hre_religion_treaty = yes
				AND = {
					hre_religion_locked = no
					hre_religion_treaty = no
				}
				AND = {
					hre_religion_locked = yes
					OR = {
						AND = {
							hre_religion = catholic
							religion = catholic
						}
						AND = {
							hre_religion = protestant
							religion = protestant
						}
						AND = {
							hre_religion = reformed
							religion = reformed
						}
						emperor = {
							religion = PREV
						}
					}
				}
			}
		}
	}
	OR = {
		NOT = { has_global_flag = hre_golden_bull_enacted }
		has_global_flag = hre_golden_bull_removed
		AND = {
			is_a_secular_country = yes
			NOT = {
				calc_true_if = {
					all_elector = {
						is_a_secular_country = yes
					}
					amount = 4
				}
			}
		}
		AND = {
			is_an_ecclesiastical_country = yes
			OR = {
				AND = {
					num_of_max_electors = { value = 8 }
					NOT = {
						calc_true_if = {
							all_elector = {
								is_an_ecclesiastical_country = yes
							}
							amount = 4
						}
					}
				}
				AND = {
					NOT = { num_of_max_electors = { value = 8 } }
					NOT = {
						calc_true_if = {
							all_elector = {
								is_an_ecclesiastical_country = yes
							}
							amount = 3
						}
					}
				}
			}
		}
	}
}
