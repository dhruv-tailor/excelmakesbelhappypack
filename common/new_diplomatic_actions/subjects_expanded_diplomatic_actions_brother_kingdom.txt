		}
		OR = {
			has_country_modifier = two_sons
			has_country_modifier = three_sons
		}
	}

		if = {
			limit = {
				can_be_overlord = brother_kingdom

				is_at_war = no
				FROM = {
					is_at_war = no
				}
				OR = {
					has_country_modifier = two_sons
					has_country_modifier = three_sons
				}
			}
			create_subject = {
				subject = FROM
				subject_type = brother_kingdom
			}

			}

		has_country_modifier = three_sons

			all_subject_country = {
				OR = {
					is_subject_of_type = personal_union

				}
			}

		FROM = {

				AND = {

					NOT = {
