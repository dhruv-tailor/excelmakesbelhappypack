update_goods = {
	set_variable = { which = $GOODS$ value = 0.00 }          	
	set_variable = { which = $GOODS$_total value = 0.00 }

	every_owned_province = { 
		limit = { 
			trade_goods = $GOODS$ 
		}

		export_to_variable = {
			variable_name = $GOODS$      
			value = modifier:trade_goods_size                         
		}	
		
		export_to_variable = {
			variable_name = tradegood_modifier     
			value = modifier:trade_goods_size_modifier                        
		}
		
		export_to_variable = {
			variable_name = tradegood_modifier_glob    
			value = modifier:global_trade_goods_size_modifier 
			who = owner 
		}
		
		change_variable = {            
			which = tradegood_modifier 
			which = tradegood_modifier_glob  
		}

		multiply_variable = {
			which = tradegood_modifier
			which = $GOODS$ 
		}

		change_variable = {            
			which = $GOODS$ 
			which = tradegood_modifier 
		}
		
		owner = {
			change_variable = {            
				which = $GOODS$ 
				which = PREV  
			}
		}
	}

	## NEW TEST STUFF
	if = {
		limit = { is_subject = no }
		change_variable = {            
			which = $GOODS$
			which = from_subjects_$GOODS$  
		}
		
		change_variable = {            
			which = $GOODS$_total 
			which = $GOODS$  
		}
	}
	else = {
		change_variable = {            
			which = $GOODS$_total 
			which = $GOODS$  
		}
		
		subtract_variable = {            
			which = $GOODS$
			which = from_subjects_$GOODS$  
		}	
	}
	##

	subtract_variable = {            
		which = $GOODS$
		which = $GOODS$_used
	}

}
