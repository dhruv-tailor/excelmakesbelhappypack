#Scripted effects relating to China, with or without DLC.

boost_eunuchs_effect = {
	if = {
		limit = {
			NOT = { has_dlc = "Mandate of Heaven" }
		}
		add_faction_influence = {
			faction = enuchs
			influence = 10
		}
	}
}
