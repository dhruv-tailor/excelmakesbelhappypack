consort_adm_bonus = {
	potential = {
		has_regency = no
		has_consort = yes
	}
	trigger = {
		variable_arithmetic_trigger = {
			custom_tooltip = consort_adm_bonus_tt
			
			export_to_variable = {
				variable_name = rulerA
				value = ADM
			}
			export_to_variable = {
				variable_name = consortA
				value = consort_adm
			}
			subtract_variable = {
				which = consortA
				which = rulerA
			}
			check_variable = {
				consortA = 2
			}
		}
	}
	monarch_admin_power = 1
}
