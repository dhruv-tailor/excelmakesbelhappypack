divorce_consort_aspect = {
	sprite = "GFX_divorce_consort_icon"
	cost = 100
	potential = {
		has_dlc = "Rights of Man"
		government = monarchy
		NOT = { has_reform = iqta }
		NOT = { has_reform = mamluk_government }
		NOT = { has_reform = ottoman_government }
	}
	trigger = {
		has_consort = yes
		is_female = no
		has_heir = no
		is_lesser_in_union = no
		has_regency = no
	}
	effect = {
		custom_tooltip = divorce_consort_tooltip
		hidden_effect = {
			remove_consort = yes
		}		
	}
	ai_will_do = {
		factor = 2
		modifier = {
			factor = 0
			NOT = {
				consort_has_personality = infertile_personality
			}
		}
		modifier = {
			factor = 0
			OR = {
				consort_adm = 2
				consort_dip = 2
				consort_mil = 2
			}
		}
	}
}
