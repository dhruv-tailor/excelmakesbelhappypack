#government_lvl_1 = {
courthouse = {
	cost = 250
	time = 12
	
	
	trigger = {
			
		OR = {
		
			#has_building = government_lvl_1
			has_building = courthouse
			
			NOT = {
				any_province_in_state = {
					
					has_construction = building
					
				}
			}
		}
		NOT = {
		OR = {
		has_province_modifier = government_lvl_1_mod
		has_province_modifier = government_lvl_2_mod
		has_province_modifier = government_lvl_3_mod
		has_province_modifier = government_lvl_4_mod
		
		}
	}
	}
	
	modifier = {
	
	local_state_maintenance_modifier = -0.25
	local_autonomy = -0.05
	local_monthly_devastation = -0.05
	#allowed_num_of_buildings = 1
		
	}
	
	on_built = {
		every_province_in_state = {
			limit = {
				NOT = { province_id = ROOT }
			}
			add_province_modifier = {
				name = government_lvl_1_mod
				duration = -1
			}
		}	
	}
	
	
	on_destroyed = {
		if = {
			limit = {
				NOT = {
					any_province_in_state = {
						has_building = government_lvl_1
					}
				}
			}
		every_province_in_state = {
				remove_province_modifier = government_lvl_1_mod
			}
		}
	}
	
	ai_will_do = {
		factor = 100
		
	}
}
