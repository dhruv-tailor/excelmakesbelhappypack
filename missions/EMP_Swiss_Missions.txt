swiss_missions_1 = {
	slot = 1
	generic = no
	ai = yes
	potential_on_load = {
		 has_dlc = "Emperor"
	}
	potential = {
		tag = SWI
		has_country_flag = use_paradox_missions
		NOT = { map_setup = map_setup_random }
		has_dlc = "Emperor"
	}
	has_country_shield = yes
	emp_swi_vaud_wallis = {
		icon = vaud_and_wallis
		required_missions = { emp_swi_start }
		position = 2
		provinces_to_highlight = {
			OR = {
				province_id = 1867
				province_id = 1871
			}
			NOT = {
				owned_by = ROOT
			}
		}
		trigger = {
			1867 = { owned_by = ROOT }
			1871 = { owned_by = ROOT }
		}
		effect = {
			1867 = { add_base_tax = 1 add_base_production = 1 add_base_manpower = 1 }
			1871 = { add_base_tax = 1 add_base_production = 1 add_base_manpower = 1 }
		}
	}
	emp_swi_geneva = {
		icon = sway_geneva
		required_missions = { emp_swi_vaud_wallis }
		position = 2
		provinces_to_highlight = {
			province_id = 4720
			OR = {
				NOT = { owned_by = ROOT }
				NOT = { is_core = ROOT }
			}
		}
		trigger = {
			SWI = {
				NOT = { 
					is_subject_of = GNV 
				}
			}
			OR = {
				4720 = { country_or_non_sovereign_subject_holds = ROOT }
				GNV = {
					is_at_war = no
					NOT = { is_subject_of = ROOT }
					has_opinion = { who = ROOT value = 100 }
					NOT = { num_of_cities = 2 }
				}
			}
		}
		effect = {
			if = {
				limit = {
					NOT = { owns = 4720 }
					GNV = {
						NOT = { is_subject_of = ROOT }
					}
				}
				custom_tooltip = swi_geneva_mission_tt
			}
			if = {
				limit = {
					GNV = {
						is_subject = no
						has_opinion = { who = ROOT value = 100 }
						NOT = { num_of_cities = 2 }
						is_at_war = no
					}
				}
				country_event = { id = emp_mission_events.2 }		
			}
			else_if = {
				limit = {
					GNV = {
						is_subject = yes
						is_at_war = no
						NOT = { is_subject_of = ROOT }
						NOT = { num_of_cities = 2 }
						has_opinion = { who = ROOT value = 100 }
					}
				}
				GNV = {
					add_liberty_desire = 100
					overlord = {
						country_event = { id = emp_mission_events.4 }
					}
				}
				support_independence_of = GNV
			}
			add_mil_power = 50
		}
	}
}
