	slot = 1
	generic = no
	ai = yes
	potential = {
		tag = HUN

	hun_poland_alliance = {
		icon = mission_noble_council
		required_missions = {  }
		position = 1

		provinces_to_highlight = {
			OR = {
				owned_by = POL
				AND = {
					NOT = { exists = POL }

					ROOT = {
						NOT = {
							num_of_owned_provinces_with = {
								value = 6
								region = poland_region
							}
						}
					}
				}
			}
		}
		
		trigger = {
			if = {
				limit = {
					exists = POL
					has_dlc = "Res Publica"
					POL = {
						has_reform = elective_monarchy
					}
				}
				OR = { 
					POL = { dynasty = root }
					AND = {
						alliance_with = POL
						POL = {
							has_opinion = {
								who = ROOT
								value = 190
							}
						}
					}
					overlord_of = POL
					AND = { 
						OR = {
							is_rival = POL
							POL = { is_rival = ROOT }
						}
						army_size = POL
					}
				}
			}
			else_if = {
				limit = { exists = POL }
				OR = {
					alliance_with = POL
					overlord_of = POL
					POL = {
						has_opinion = {
							who = ROOT
							value = 190
						}
					}
					AND = { 
						OR = {
							is_rival = POL
							POL = { is_rival = ROOT }
						}
						army_size = POL
					}
				}
			}
			else_if = {
				limit = {
					NOT = { exists = POL }
				}
				OR = {
					army_size_percentage = 1
					num_of_owned_provinces_with = {
						value = 6
						region = poland_region
					}
				}
			}
			is_subject = no
		}
		
		effect = {
			if = {
				limit = {
					OR = {
						NOT = { alliance_with = POL }
						NOT = {
							moldavia_area = {
								type = all
								country_or_non_sovereign_subject_holds = POL
							}
						}
					}
				}
				moldavia_area = {
					limit = {
						NOT = { is_core = ROOT }
						NOT = { is_permanent_claim = ROOT }
					}
					add_permanent_claim = ROOT
				}
			}
			else = {
				add_dip_power = 100
			}
			wallachia_area = {
				limit = {
					NOT = { is_core = ROOT }
					NOT = { is_permanent_claim = ROOT }
				}
				add_permanent_claim = ROOT
			}
		}
	}
	
	hun_vassal_states = {
		icon = mission_unite_home_region
		required_missions = { hun_poland_alliance }

	}

		icon = mission_eastern_european_city
		required_missions = { hun_vassal_states }
		position = 3
		
		provinces_to_highlight = {
			OR = {
				area = serbia_area
				area = rascia_area
			}
			NOT = { country_or_non_sovereign_subject_holds = ROOT }
			ROOT = {
				NOT = {
					num_of_provinces_owned_or_owned_by_non_sovereign_subjects_with = {
						value = 8

							area = serbia_area
							area = rascia_area
						}
					}
				}
			}
		}
		
		trigger = {
			num_of_provinces_owned_or_owned_by_non_sovereign_subjects_with = {
				value = 8
				OR = {
					area = serbia_area
					area = rascia_area

			}

		}

					name = "hun_moldavia_mod"

				}
			}

				}
			}
			croatia_area = {
				limit = {
					NOT = { is_core = ROOT }

				}
				add_permanent_claim = ROOT
			}
