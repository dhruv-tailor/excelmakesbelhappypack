			}
		}
		trigger = {

				}
			}
		}
		effect = {
			add_mil_power = 100
			add_legitimacy_equivalent = { value = 10 }
			4131 = {
				if = {
					limit = {
						is_expanded_mod_active = { mod = trade_goods }
						owned_by = ROOT
						fort_level = 2
					}
					add_province_modifier = {
						name = uesugi_secure_the_northern_border_province_modifier
						duration = -1
					}
				}
			}
			hokkaido_area = {
				limit = {
					NOT = { owned_by = ROOT }
					NOT = { is_core = ROOT }
				}
				add_permanent_claim = ROOT
			}
		}
	}
	
	uesugi_conquer_ainu = {

		required_missions = { uesugi_secure_the_northern_border }

					}
				}
			}
			1032 = {
				if = {
