		icon = mission_indian_stateman
		required_missions = { }
		provinces_to_highlight = {
			OR = {
				owned_by = TID
				owned_by = TER
			}
			NOT = { owned_by = ROOT }
		}
		trigger = {
			if = {
				limit = {
					tag = TID
					NOT = {
						exists = TER
					}
				}
			}
			else_if = {
				limit = {
					tag = TER
					NOT = {
						exists = TID
					}
				}
			}
			else_if = {
				limit = {
					tag = TID
					any_province = {
						owned_by = TER
						is_colony = yes
					}
				}
				649 = {
					owned_by = TID
				}
			}
			else_if = {
				limit = {
					tag = TER
					any_province = {
						owned_by = TID
						is_colony = yes
					}
				}
				650 = {
					owned_by = TER
				}
			}
			else = {
				NOT = { exists = TID }
				NOT = { exists = TER }
			}
		}
		effect = {
			random_owned_province = {
				province_event = {
					id = flavor_ter.1
				}
			}
		}
	}
	spice_colony_1 = {
		icon = mission_market_place_with_asian_traders
		required_missions = { spice_defeat_rival }
		trigger = {
			any_owned_province = {
				is_city = yes

					tooltip = spice_island_colony_tt
					has_province_flag = spice_island_colony_flag

			}
		}
		effect = {

					id = flavor_ter.1
				}
			}
		}
	}
	spice_colony_2 = {
